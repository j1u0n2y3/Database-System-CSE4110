create table house(address    varchar(100),zip_code    varchar(10) not null,primary key (address));
create table account(account_num    varchar(25),depositor_name  varchar(20) not null,primary key (account_num));
create table type(weight int,box varchar(10) not null,primary key (weight));
create table transportation(trans_ID   varchar(20),start_date  datetime,end_date    datetime,primary key (trans_ID, start_date),check (start_date<=end_date));
create table trans_type(trans_ID   varchar(20),trans  varchar(10) not null,primary key (trans_ID),check (trans in ('warehouse', 'airplane', 'truck')));
create table customer(ID varchar(20),address varchar(100) not null,name    varchar(20) not null,primary key (ID),foreign key (address) references house (address));
create table contract_info(ID varchar(20),account_num varchar(25),primary key (ID),foreign key (ID) references customer (ID),foreign key (account_num) references account (account_num));
create table package(package_ID varchar(20),paid    int,weight  int,timeliness  int,primary key (package_ID),foreign key (weight) references type (weight),check (timeliness in (12, 24, 48, 72, 96, 120)));
create table receiver(package_ID varchar(20),ID  varchar(20),primary key (package_ID),foreign key (package_ID) references package (package_ID),foreign key (ID) references customer (ID));
create table sender(package_ID varchar(20),ID  varchar(20),role    varchar(10),primary key (package_ID),foreign key (package_ID) references package (package_ID),foreign key (ID) references customer (ID),check (role in ('customer', 'company', 'prepaid')));
create table object(package_ID varchar(20),value   int,primary key (package_ID),foreign key (package_ID) references package (package_ID));
create table obj_content(package_ID varchar(20),content varchar(20),primary key (package_ID, content));
create table trans_by(package_ID varchar(20),trans_ID    varchar(20),start_date  datetime,primary key (package_ID, trans_ID, start_date),foreign key (package_ID) references package (package_ID),foreign key (trans_ID, start_date) references transportation (trans_ID, start_date));
